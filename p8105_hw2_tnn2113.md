subway_data = 
  read_csv("./data/NYC_Transit_Subway_Entrance_And_Exit_Data.csv") %>%

  select(-(ada_notes:entrance_location))  %>% 
  select(-c("division", "exit_only", "staffing", "staff_hours")) %>% 
  mutate(entry = recode(entry, `YES` = "TRUE", `NO` = "FALSE"))
